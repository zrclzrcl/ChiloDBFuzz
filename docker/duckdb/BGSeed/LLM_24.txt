CREATE TABLE integers (
    id INTEGER PRIMARY KEY,
    value INTEGER
);

CREATE VIEW large_integers AS
SELECT value FROM integers WHERE value > 1;

-- Test DuckDB-specific feature: APPROXIMATE PERCENTILE
SELECT approx_percentile(0.5) WITHIN GROUP (ORDER BY value) FROM integers;
