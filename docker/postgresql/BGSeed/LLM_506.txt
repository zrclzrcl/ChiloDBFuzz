-- Sample initial seed for a DBMS fuzzer

-- Create a simple table
CREATE TABLE employees (
    id INT PRIMARY KEY,
    name VARCHAR(255),
    salary DECIMAL(10, 2)
);

-- Insert some data
INSERT INTO employees (id, name, salary) VALUES (1, 'Alice', 50000.00);
INSERT INTO employees (id, name, salary) VALUES (2, 'Bob', 60000.00);

-- Select all employees
SELECT * FROM employees;

-- Update an employee's salary
UPDATE employees SET salary = 55000.00 WHERE id = 1;

-- Delete an employee
DELETE FROM employees WHERE id = 2;

-- Select with a WHERE clause
SELECT name FROM employees WHERE salary > 52000.00;

-- Drop the table
DROP TABLE employees;
