-- Sample initial seed for a DBMS fuzz tester

-- Create a simple table
CREATE TABLE employees (
    id INT PRIMARY KEY,
    name VARCHAR(50),
    salary DECIMAL(10, 2)
);

-- Insert some data
INSERT INTO employees (id, name, salary) VALUES (1, 'Alice', 60000.00);
INSERT INTO employees (id, name, salary) VALUES (2, 'Bob', 75000.00);
INSERT INTO employees (id, name, salary) VALUES (3, 'Charlie', 90000.00);

-- Select all employees
SELECT * FROM employees;

-- Update an employee's salary
UPDATE employees SET salary = 80000.00 WHERE id = 2;

-- Delete an employee
DELETE FROM employees WHERE id = 1;

-- Select employees with salary greater than a certain value
SELECT * FROM employees WHERE salary > 70000.00;

-- Drop the table
DROP TABLE employees;
